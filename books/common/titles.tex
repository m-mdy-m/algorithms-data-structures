\usepackage{titlesec}              % Title formatting control

% Chapter spacing
\titlespacing*{\chapter}
{0pt}               % left margin
{0pt}               % space before
{2.5ex plus 0.5ex}  % space after

% Section spacing
\titlespacing*{\section}
{0pt}               % left margin
{2ex plus 0.4ex minus 0.2ex} % space before
{1.5ex plus 0.3ex minus 0.1ex} % space after

% Subsection spacing
\titlespacing*{\subsection}
{0pt}               % left margin
{1.5ex plus 0.3ex minus 0.1ex} % space before
{1ex plus 0.2ex minus 0.1ex}   % space after

% Subsubsection spacing
\titlespacing*{\subsubsection}
{0pt}               % left margin
{1ex plus 0.2ex minus 0.1ex}   % space before
{0.8ex plus 0.1ex minus 0.1ex} % space after
